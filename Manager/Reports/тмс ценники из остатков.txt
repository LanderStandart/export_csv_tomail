function get_report_id;
var report_id: Integer;
    q: TIBQuery;
begin
  q:=dm.TempQuery(nil);
  q.Active:=false;
  q.SQL.Text:='select first 1 id from reports where id=-1';
  q.Active:=true;
  report_id:=q.FieldByName('id').AsInteger;
  if report_id=0 then
  begin
    q.Active:=false;
    q.SQL:='insert into reports(ID,PARENT_ID,STATUS,REPORTTYPE,SORTING,CAPTION) values '+
    ' (-1,0,0,5,0,''Объединенные ценники'')';
    q.ExecSQL;
    //Добавиьт еще загрузку с сервера последней версии
  end;
  q.Active:=false;
  q.SQL.Text:='select first 1 id from reports where id=-1';
  q.Active:=true;
  report_id:=q.FieldByName('id').AsInteger;
  result:=report_id;
end;
